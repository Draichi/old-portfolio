	export default class JobService {
		constructor (resource) {
			this._resource = resource('jobs{/id}')
		}
	// method to add a new job
		addNew (job) {
			// update if already has id
			if (job._id) {
				return this._resource
					.update({id: job._id}, job)
			} else {
			// if not, add new
				return this._resource
					.save(job)
			}
		}
	}
