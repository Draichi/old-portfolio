	// This is a sample of a vue.js component
	export default {
		components: {
			'meu-painel': Painel,
			'imagem-responsiva': ImagemResponsiva,
			'meu-botao': Botao
		},
		data() {
			return {
			titulo: 'Projeto visa o uso de diretivas',		
			}
		},
		methods:{
			remove(foto){
				this.service.apaga(foto._id)
					.then(() => {
						let indice = this.fotos.indexOf(foto);
						this.fotos.splice(indice, 1);
						this.mensagem = 'Foto removida com sucesso';
					}, err => this.mensagem = 'Não foi possivel remover');   	
			}
		}
		computed: { 
			fotosComFiltro() {
				if(this.filtro) {
					let exp = new RegExp(this.filtro.trim(),'i');
						return this.fotos.filter(foto => exp.test(foto.titulo));
				} else {
					return this.fotos;
				}
			}
		}
